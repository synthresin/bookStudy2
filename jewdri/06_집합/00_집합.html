집합?

  • 집합은 정렬되지않은 컬렉션으로 원소는 반복되지 않는다.
  • 집합자료구조는 수학책에 나오는 유한집합(finite set)의 개념을 컴퓨터과학에 적용한 것
  • 집합은 정렬개념이 없는, 원소가 중복되지 안흔ㄴ 배열이라고 볼 수 있다.

집합 만들기

리스트 끝에 원소 추가하기

배열이 아닌 객체로 집합(원소)을 표현한다.
function Set(){
   this.items = {};
}

  • add(원소) : 원소를 추가한다.
  • remove(원소) : 원소를 삭제한다.
  • has(원소) : 어떤 원소가 집합에 포함되어 있는지 여부를 true/false로 반환한다.
  • clear() : 모든 원소를 삭제한다.
  • size(원소) : 원소를 삭제한다.
  • values() : 집합의 모든 원소를 배열 형태로 반환한다.

has(원소) 메소드

add 메소드

Set.prototype.add = function(value){
    if(!this.has(value)){
        this.items[value] = value;
        return true;
    }
    return false;
}

  • 3가지 구현방법이 있다.

- 자바스크립트 Object클래스에 이미 내장된 함수를 이용

집합연산

집합에서 사용 가능한 연산은 네가지가 있다.
  • 합집합 : 두 집합 중 어느 한쪽이라도 포함된 원소로 구성된 집합을 구한다.
  • 교집합 : 두 집합 모두 포함되어 있는 원소로 구성된 집합을 구한다.
  • 차집합 : 첫 번째 집합에는 있지만 두 번째 집합에는 없는 원소로 구성된 집합을 구한다.
  • 부분집합 : 어떤 집합이 다른 집합의 일부인지 확인한다.

합집합(Union)

  • 합집합은 수학기호로 A∪B와 같이 표시
  • A, B 어느 한쪽에 속하는 원소 x의 집합
Set.prototype.union = function(otherSet){
    var unionSet = new Set();
    var values =  this.values();
    for(var i=0; i<values.length; i++){
        unionSet.add(values[i]);
    }
    values =  otherSet.values();
     for(var i=0; i<values.length; i++){
        unionSet.add(values[i]);
    }
    return unionSet;
}

교집합(intersection)

  • 교집합은 수학기호로 A∩B와 같이 표시
  • A에도 속하고 B에도 속하는 원소 x의 집합
Set.prototype.intersection = function(otherSet){
    var intersectionSet = new Set();
    var values = this.values();
    for(var i=0; i<values.length; i++){
        if(otherSet.has(values[i])){
            intersectionSet.add(values[i]);
        }
    }
    return intersectionSet;
}

차집합(difference)

  • 차집합은 수학기호로 A-B와 같이 표시
  • A에는 속하지만 B에는 속하지 않는 원소 x의 집합
Set.prototype.difference = function(otherSet){
    var differenceSet = new Set();
    var values = this.values();
    for(var i=0; i<values.length; i++){
        if(!otherSet.has(values[i])){
            differenceSet.add(values[i]);
        }
    }
    return differenceSet;
}

부분집합(subset)

  • 부분집합은 수학기호로 A⊂B와 같이 표시
  • A의 모든 원소는 반드시 B에 존재해야 함을 의미
Set.prototype.subset = function(otherSet){
    if(this.size() > otherSet.size()){
        return false;
    }else{
        var values = this.values();
        for(var i=0; i<values.length; i++){
            if(!otherSet.has(values[i])){
                return false;
            }
        }
        return true;
    }
}
